\section{Giới thiệu}

\begin{frame}{Tổng quan}
\begin{itemize}
\item ChaCha20: Mã hóa dòng hiện đại
\begin{itemize}
\item Thiết kế bởi Daniel J. Bernstein (2008)
\item Cải tiến từ Salsa20
\end{itemize}
\item AES-CTR: Mã hóa khối ở chế độ Counter
\begin{itemize}
\item Tiêu chuẩn mã hóa tiên tiến (AES)
\item Chế độ CTR biến AES thành mã hóa dòng
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Phân tích ChaCha20}
\begin{itemize}
\item Ưu điểm:
\begin{itemize}
\item Hiệu năng tốt với dữ liệu lớn
\item Không yêu cầu bảng tra cứu
\item Dễ cài đặt và tối ưu
\end{itemize}
\item Nhược điểm:
\begin{itemize}
\item Hiệu năng không ổn định với file nhỏ
\item Không có hỗ trợ phần cứng phổ biến
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Phân tích AES-CTR}
\begin{itemize}
\item Ưu điểm:
\begin{itemize}
\item Hiệu năng ổn định
\item Hỗ trợ phần cứng (AES-NI)
\item Tiêu chuẩn được công nhận
\end{itemize}
\item Nhược điểm:
\begin{itemize}
\item Yêu cầu bảng tra cứu S-box
\item Phụ thuộc vào hỗ trợ phần cứng
\end{itemize}
\end{itemize}
\end{frame}

\section{Kết luận}

\begin{frame}{Khuyến nghị sử dụng}
\begin{itemize}
\item Sử dụng ChaCha20 khi:
\begin{itemize}
\item Xử lý file lớn
\item Không có hỗ trợ phần cứng đặc biệt
\item Cần hiệu năng cao trên phần cứng đa dạng
\end{itemize}
\item Sử dụng AES-CTR khi:
\begin{itemize}
\item Cần hiệu năng ổn định
\item Có hỗ trợ phần cứng AES
\item Tuân thủ tiêu chuẩn bắt buộc
\end{itemize}
\end{itemize}
\end{frame}
